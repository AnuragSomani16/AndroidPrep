1. ViewModel — the cornerstone

ViewModel = a lifecycle-aware container of UI state that survives configuration changes (rotation, theme switch). It's stored in the ViewModelStore attached to the Activity / Fragment / Nav backstack entry. Cleared in onCleared() when the owner is permanently gone. Does NOT survive process death — for that, use SavedStateHandle (Bundle-backed) injected into the ViewModel.

ViewModel survives configuration changes (rotation) but is cleared when the user explicitly leaves the screen (onCleared()). It does NOT survive process death — for that you need SavedStateHandle.

@HiltViewModel
class HomeVm @Inject constructor(
    private val savedState: SavedStateHandle,    // survives process death (Bundle-based)
    private val getFeed: GetFeedUseCase
) : ViewModel() {

    // Preserved across process death
    val query: StateFlow<String> = savedState.getStateFlow("query", "")

    val results: StateFlow<List<Item>> = query
        .debounce(300)
        .flatMapLatest { q -> getFeed(q) }
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

    fun setQuery(q: String) { savedState["query"] = q }

    override fun onCleared() {
        super.onCleared()
        // viewModelScope auto-cancels — usually nothing to do here
    }
}
When to use SavedStateHandle

For state you'd be sad to lose if Android killed your process while backgrounded — search input, scroll position, multi-step form progress. Bundle-sized data only (a few KB). For larger state, persist to Room or DataStore.

2. Room — full playlist database example

Schema design — playlists with songs (many-to-many)

// Entity 1 — playlist
@Entity(tableName = "playlists")
data class PlaylistEntity(
    @PrimaryKey val id: String,
    val name: String,
    val createdAt: Long,
    val updatedAt: Long,
    val coverUrl: String?,
    val isPublic: Boolean
)

// Entity 2 — song (canonical record)
@Entity(tableName = "songs")
data class SongEntity(
    @PrimaryKey val id: String,
    val title: String,
    val artistId: String,
    val durationMs: Long,
    val audioUrl: String
)

// Cross-reference table (many-to-many)
@Entity(
    tableName = "playlist_songs",
    primaryKeys = ["playlistId", "songId"],
    foreignKeys = [
        ForeignKey(entity = PlaylistEntity::class, parentColumns = ["id"], childColumns = ["playlistId"], onDelete = ForeignKey.CASCADE),
        ForeignKey(entity = SongEntity::class,    parentColumns = ["id"], childColumns = ["songId"],    onDelete = ForeignKey.CASCADE)
    ],
    indices = [Index("songId")]
)
data class PlaylistSongCrossRef(
    val playlistId: String,
    val songId: String,
    val position: Int           // ordering within playlist
)

// Composed model — playlist with its songs
data class PlaylistWithSongs(
    @Embedded val playlist: PlaylistEntity,
    @Relation(
        parentColumn = "id",
        entityColumn = "id",
        associateBy = Junction(
            value = PlaylistSongCrossRef::class,
            parentColumn = "playlistId",
            entityColumn = "songId"
        )
    )
    val songs: List<SongEntity>
)

DAO — Flow-based, suspend, transactional

@Dao
interface PlaylistDao {
    @Query("SELECT * FROM playlists ORDER BY updatedAt DESC")
    fun observeAll(): Flow<List<PlaylistEntity>>

    @Transaction
    @Query("SELECT * FROM playlists WHERE id = :id")
    fun observeWithSongs(id: String): Flow<PlaylistWithSongs?>

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun upsertPlaylist(playlist: PlaylistEntity)

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun upsertSongs(songs: List<SongEntity>)

    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun upsertCrossRefs(refs: List<PlaylistSongCrossRef>)

    @Query("DELETE FROM playlist_songs WHERE playlistId = :playlistId AND songId = :songId")
    suspend fun removeSong(playlistId: String, songId: String)

    @Query("UPDATE playlist_songs SET position = :pos WHERE playlistId = :p AND songId = :s")
    suspend fun reorder(p: String, s: String, pos: Int)

    @Transaction
    suspend fun addSongToPlaylist(playlistId: String, song: SongEntity, position: Int) {
        upsertSongs(listOf(song))
        upsertCrossRefs(listOf(PlaylistSongCrossRef(playlistId, song.id, position)))
        upsertPlaylist(getPlaylist(playlistId).copy(updatedAt = System.currentTimeMillis()))
    }

    @Query("SELECT * FROM playlists WHERE id = :id")
    suspend fun getPlaylist(id: String): PlaylistEntity
}

Database with TypeConverters and Migrations

class Converters {
    @TypeConverter fun fromList(list: List<String>): String = list.joinToString("")
    @TypeConverter fun toList(s: String): List<String> = if (s.isEmpty()) emptyList() else s.split("")
}

@Database(
    entities = [PlaylistEntity::class, SongEntity::class, PlaylistSongCrossRef::class],
    version = 2,
    exportSchema = true
)
@TypeConverters(Converters::class)
abstract class AppDb : RoomDatabase() {
    abstract fun playlists(): PlaylistDao
}

// Migration when you add a column
val MIGRATION_1_2 = object : Migration(1, 2) {
    override fun migrate(db: SupportSQLiteDatabase) {
        db.execSQL("ALTER TABLE playlists ADD COLUMN isPublic INTEGER NOT NULL DEFAULT 0")
    }
}

@Module
@InstallIn(SingletonComponent::class)
object DbModule {
    @Provides @Singleton
    fun provideDb(@ApplicationContext ctx: Context): AppDb =
        Room.databaseBuilder(ctx, AppDb::class.java, "app.db")
            .addMigrations(MIGRATION_1_2)
            .build()

    @Provides fun playlistDao(db: AppDb): PlaylistDao = db.playlists()
}
Migration discipline

Never use fallbackToDestructiveMigration() in production. Always write a migration. Set exportSchema = true so Room generates a JSON schema per version that you check into Git — this is your migration test fixture.

Senior probes on Room

Q. Why Room over plain SQLite?

Compile-time SQL verification, type-safe queries, automatic Flow emission on table change, transaction support via annotations, type converters, migration helpers, schema export for migration testing. Plain SQLite gives you string queries and runtime crashes when the schema drifts.

Q. Single-table vs normalized — when?

Normalized for write-heavy / shared-across-screens (songs referenced by many playlists). Single-table denormalized for read-heavy, view-specific data (a "feed item" with embedded user info that rarely changes). For mobile apps, normalize moderately — DB size is small, joins are fast.

Q. How do you observe joined data?

@Transaction + @Relation on a composed data class returned from a @Query that returns a Flow. The transaction ensures you see a consistent snapshot.

Q. What happens with @Query that returns Flow?

Room's invalidation tracker registers your query against the tables it touches. When any of those tables change, Room re-runs the query and emits a new value into the Flow. Cheaper than a manual cursor loop and lifecycle-safe.

// proto/user_prefs.proto
syntax = "proto3";
option java_multiple_files = true;
message UserPrefs {
  string display_name = 1;
  bool dark_mode = 2;
  int32 max_quality = 3;
}
object UserPrefsSerializer : Serializer<UserPrefs> {
    override val defaultValue: UserPrefs = UserPrefs.getDefaultInstance()
    override suspend fun readFrom(input: InputStream) = try { UserPrefs.parseFrom(input) }

    override suspend fun writeTo(t: UserPrefs, output: OutputStream) = t.writeTo(output)

6. Paging 3 — for big lists

Network-only (PagingSource)

class FeedPagingSource(private val api: FeedApi) : PagingSource<Int, FeedItem>() {
    override suspend fun load(params: LoadParams<Int>): LoadResult<Int, FeedItem> = try {
        val page = params.key ?: 0
        val response = api.feed(page = page, limit = params.loadSize)
        LoadResult.Page(
            data = response.items,
            prevKey = if (page == 0) null else page - 1,
            nextKey = if (response.items.isEmpty()) null else page + 1
        )
    } catch (e: IOException) { LoadResult.Error(e) }
    override fun getRefreshKey(state: PagingState<Int, FeedItem>) = null
}

val pager = Pager(
    config = PagingConfig(pageSize = 20, prefetchDistance = 5),
    pagingSourceFactory = { FeedPagingSource(api) }
).flow.cachedIn(viewModelScope)

Network + DB (RemoteMediator) — offline-first paging

@OptIn(ExperimentalPagingApi::class)
class FeedRemoteMediator(
    private val api: FeedApi,
    private val db: AppDb
) : RemoteMediator<Int, FeedItemEntity>() {
    override suspend fun load(loadType: LoadType, state: PagingState<Int, FeedItemEntity>): MediatorResult {
        return try {
            val page = when (loadType) {
                LoadType.REFRESH -> 0
                LoadType.PREPEND -> return MediatorResult.Success(endOfPaginationReached = true)
                LoadType.APPEND -> (state.lastItemOrNull()?.page ?: 0) + 1
            }
            val response = api.feed(page, state.config.pageSize)
            db.withTransaction {
                if (loadType == LoadType.REFRESH) db.feedDao().clear()
                db.feedDao().insertAll(response.items.map { it.toEntity(page) })
            }
            MediatorResult.Success(endOfPaginationReached = response.items.isEmpty())
        } catch (e: IOException) { MediatorResult.Error(e) }
    }
}

val pager = Pager(
    config = PagingConfig(pageSize = 20),
    remoteMediator = FeedRemoteMediator(api, db),
    pagingSourceFactory = { db.feedDao().pagingSource() }
).flow

In-memory LRU

class UserCache @Inject constructor() {
    private val cache = object : LruCache<String, User>(100) {
        override fun sizeOf(key: String, value: User): Int = 1   // count entries; for bitmaps use bytes
    }
    fun get(id: String): User? = cache[id]
    fun put(user: User) { cache.put(user.id, user) }
    fun clear() = cache.evictAll()
}

Cache-aside pattern (read-through with fallback)

class UserRepo @Inject constructor(
    private val api: UserApi,
    private val dao: UserDao,
    private val mem: UserCache
) {
    suspend fun get(id: String): User =
        mem.get(id)
            ?: dao.byId(id)?.toDomain()?.also { mem.put(it) }
            ?: api.fetch(id).toDomain().also {
                dao.upsert(it.toEntity())
                mem.put(it)
            }
}
Reads cascade from fastest to slowest:
  1. Memory cache hit — mem.get(id) returns immediately if present. ~microseconds.
  2. DB hit — if memory misses, dao.byId(id) reads from disk. ~milliseconds. ?.also { mem.put(it) } warms the memory cache for next time.
  3. Network — if both miss, api.fetch(id) goes over the wire. ~hundreds of ms. We write through to both caches before returning so subsequent reads are fast.
The Elvis chain (?:) makes the fallback flow read top-to-bottom. If you wanted offline-first instead, you'd observe the DAO's Flow and refresh from network in the background — see Scenario Drills #2.

Read-through (Flow-based offline-first)

fun observe(id: String): Flow<User> = dao.observe(id)
    .map { it?.toDomain() ?: api.fetch(id).also { dao.upsert(it.toEntity()) }.toDomain() }
    .distinctUntilChanged()

8. Coil — image caching in Compose

// In Application — global config
class App : Application(), ImageLoaderFactory {
    override fun newImageLoader(): ImageLoader = ImageLoader.Builder(this)
        .memoryCache {
            MemoryCache.Builder(this)
                .maxSizePercent(0.25)              // 25% of app heap
                .build()
        }
        .diskCache {
            DiskCache.Builder()
                .directory(cacheDir.resolve("image_cache"))
                .maxSizeBytes(100 * 1024 * 1024)   // 100MB
                .build()
        }
        .respectCacheHeaders(false)                 // ignore weak HTTP cache headers
        .crossfade(true)
        .build()
}

Senior phrase

"Coil's two-tier cache (memory LruBitmapPool + disk LRU) means rapid scrolling re-uses bitmaps without decoding; the disk cache survives process death. I cap memory at 25% of heap on Android to leave room for the rest of the app on low-end devices."
